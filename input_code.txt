// Inicia la carrera
load speed = 0;
load max_speed = 300;
turbo boost = 100;

// Comienza la primera vuelta
lap (i = 0; i < 5; i = i + 1) {
check (speed < max_speed) {
speed = speed + 50;
shout ("Acelerando: " + speed + " km/h");
}
}

// Usar nitro al mÃ¡ximo
move activateNitro() {
shout ("Nitro activado!");
speed = speed + boost;
return speed;
}
